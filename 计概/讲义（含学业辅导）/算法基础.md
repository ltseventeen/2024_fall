def Euler(n):
    prime = []
    is_prime = [False] * (n+1)
    is_prime[0] = is_prime[1] = False
    for i in range(2, n + 1):
        is_prime[i] = True
    for i in range(2, n + 1):
        if is_prime[i]:
            prime.append(i)
        for pri_j in prime:
            if i * pri_j > n:
                break
            is_prime[i * pri_j] = False
            if i % pri_j == 0:
                break
    return prime
